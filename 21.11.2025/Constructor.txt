package nov21;
	class Student {
	    String name;
	    int age;

	   
	    Student(String n, int a) {
	        name = n;
	        age = a;
	    }

	    void display() {
	        System.out.println("Name: " + name + ", Age: " + age);
	    }
	}

	public class Constructor {
	    public static void main(String[] args) {
	        Student s1 = new Student("Arun", 18);
	        Student s2 = new Student("Bala", 19);

	        s1.display();
	        s2.display();
	    }
	}
