package nov21;

class Calculator {

    int add(int a, int b) {
        return a + b;
    }

    int add(int a, int b, int c) {
        return a + b + c;
    }

    double add(double a, double b) {
        return a + b;
    }
}

public class overloading {
    public static void main(String[] args) {
        Calculator obj = new Calculator();
        int a = 10;
        int b = 20;
        int c = 30;

        System.out.println(obj.add(a, b,c));
    }
}
