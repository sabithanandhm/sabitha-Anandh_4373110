package nov17;

public class OMRking {
    public static void main(String[] args) {

        Thread t1 = new Thread(new Son1Thread());
        Thread t2 = new Thread(new Son2Thread());

        t1.start();   
        t2.start();    
    }
}


class GrandFather {
    void oldBike() {
        System.out.println("Royal Enfield 1983");
    }
}

class Father extends GrandFather {
    void oldBike() {
        System.out.println("Royal Enfield 2002");
    }
}

class Son1 extends Father {
    void bike() {
        System.out.println("Ktm 390");
    }
}

class Son2 extends Father {
    void bike() {
        System.out.println("RC KTM 390");
    }
}


class Son1Thread implements Runnable {
    public void run() {
        Son1 s1 = new Son1();
        s1.bike();
    }
}

class Son2Thread implements Runnable {
    public void run() {
        Son2 s2 = new Son2();
        s2.bike();
        s2.oldBike();
    }
}
